bool nextToABox()
{
    bool result = false;
    int  k = 0;
    int  c;

    ref_world v_world = world_now;
    ref_robot v_robot = robot_now;

    if ( v_world != NULL &&
         v_robot != NULL &&
         checkRobotStatus ( ) )
    {
       do
       {
           k = k + 1;
           c = v_world->data [ v_robot->street-1 ][ v_robot->avenue-1 ][ k ];
       }
       while ( k+1 < v_world->depth &&
               ! ( 'a' <= c && c <= 'z') );

       result = ( 'a' <= c && c <= 'z' );
    }

    return ( result );
} // end nextToABox ( )
