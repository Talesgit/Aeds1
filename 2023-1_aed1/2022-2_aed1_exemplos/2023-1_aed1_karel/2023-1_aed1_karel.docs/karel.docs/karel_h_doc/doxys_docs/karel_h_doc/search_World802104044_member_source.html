int search_World(int avenue, int street, char object)
{
  int  result = -1;
  int  z      = -1;
  char c;

  ref_world v_world = world_now;

  if ( v_world != NULL &&
       areValid ( avenue, street ) )
  {
       do
       {
          z = z + 1;
          c = v_world->data [ street-1 ][ avenue-1 ][ z ];
       }
       while ( z < v_world->depth-1 && c != object );
       if ( c == object )
          result = z;
  }
  return ( result );
} // end search_World ( )
