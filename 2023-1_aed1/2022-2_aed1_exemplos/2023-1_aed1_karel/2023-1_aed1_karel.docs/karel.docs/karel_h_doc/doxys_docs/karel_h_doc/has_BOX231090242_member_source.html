int has_BOX(int avenue, int street)
{
  int  result = 0;
  int  z = -1;
  char c;

  ref_world v_world = world_now;

  if ( v_world != NULL &&
       areValid ( avenue, street ) )
  {
       do
       {
          z = z + 1;
          c = v_world->data [ street-1 ][ avenue-1 ][ z ];
          if ( 'a' <= c  && c <= 'z' )
          {
             result = c;
          }
       }
       while ( z < v_world->depth-1 );
  }
  return ( result );
} // end has_BOX ( )
