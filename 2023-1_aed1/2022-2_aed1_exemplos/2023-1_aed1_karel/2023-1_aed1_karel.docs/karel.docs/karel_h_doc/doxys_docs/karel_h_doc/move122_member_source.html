void move()
{
    ref_robot v_robot = robot_now;

    if ( checkRobotStatus ( ) &&
         frontIsClear ( ) )
    {
        switch ( v_robot->direction )
        {
            case NORTH  :
              set_Robot (  0, +1 );
              break;
            case EAST  :
              set_Robot ( +1,  0 );
              break;
            case SOUTH :
              set_Robot (  0, -1 );
              break;
            case WEST  :
              set_Robot ( -1,  0 );
              break;
        }
    }
    else
    {
        show_Error ( "ERROR: Can't move." );
    }
} // end move ( )
