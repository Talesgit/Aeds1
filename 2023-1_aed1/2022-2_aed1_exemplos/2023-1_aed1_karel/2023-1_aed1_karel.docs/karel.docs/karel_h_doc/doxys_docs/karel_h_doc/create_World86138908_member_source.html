void create_World(char *name)
{
  ref_world v_world = world_now;

  if ( v_world != NULL )
  {
     v_world->height  = HEIGHT;
     v_world->width   = WIDTH ;
     v_world->depth   = DEPTH ;

     if ( name != NULL )
     {
        strcpy ( v_world->name, name );
     }
     else
     {
        strcpy ( v_world->name, "Karel Robot - v0.3" );
     }
     reset_World ( );
  }
} // end create_World ( )
