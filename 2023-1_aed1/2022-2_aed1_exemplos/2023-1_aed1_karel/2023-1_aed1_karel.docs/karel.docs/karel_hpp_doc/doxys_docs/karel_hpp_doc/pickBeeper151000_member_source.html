void Robot::pickBeeper()
{
    int  x;

    if ( world != nullptr )
    {
         x = world->has_BEEPERS ( avenue, street );
         if ( x <= 0 )
         {
            show_Error ( "ERROR: No beeper." );
         }
         else
         {
            world->remove ( avenue, street, BEEPER );
            beepers = beepers + 1;
            debug ( );
         }
    }
} // end Robot::pickBeeper ( )
