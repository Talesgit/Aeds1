void World::read(const char *fileName)
{
  FILE* f_world;

  int   x, y, z;
  char  c;
  char  name[80];

  create ( "Karel_Robot_-_v0.1" );

  f_world = fopen ( fileName, "rt" );

  if ( f_world )
  {
     fscanf  ( f_world, "%s",
               name  );
     fscanf  ( f_world, "%d %d %d",
               &x, &y, &z  );
     if ( 0 < x && x <= HEIGHT &&
          0 < y && y <= WIDTH  &&
          0 < z && z <= DEPTH  )
     {
      // set world boundaries
         height = x;
         width  = y;
         depth  = z;
      // set world name
         if ( name != nullptr )
         {
            strcpy ( this->name, name );
         }
      // read data and fill world slots
         fscanf ( f_world, "%d %d %d %c", &x, &y, &z, &c );
         while ( ! feof ( f_world ) )
         {
               if ( 0 <= x && x < height &&
                    0 <= y && y < width  &&
                    0 <= z && z < depth   )
               {
                  data[x][y][z] = c;
               }
               fscanf ( f_world, "%d %d %d %c", &x, &y, &z, &c );
         }
     }
     fclose ( f_world );
  }
} // end World::read ( )
