bool Robot::nextToARobot()
{
    bool result = false;
    char c;
    int  k=0;

    if ( world != nullptr &&
         checkStatus ( )  )
    {
       do
       {
           k = k + 1;
           c = world->data [ street-1 ][ avenue-1 ][ k ];
       }

               ! ('A' <= c && c <= 'Z' ) );
