void Robot::set(int x, int y)
{
    int  z;

    if ( checkStatus ( ) &&
       ( 1 <= avenue+x && avenue+x <= world->width  &&
         1 <= street+y && street+y <= world->height ) )
    {
       z = world->has_slot ( avenue+x, street+y );
       if ( z >= 0 )
       {
           world->remove ( avenue, street, symbol );
           avenue = avenue+x;
           street = street+y;
           world->set    ( avenue, street, symbol );
       }
       debug ( );
    }
} // end Robot::set ( )
