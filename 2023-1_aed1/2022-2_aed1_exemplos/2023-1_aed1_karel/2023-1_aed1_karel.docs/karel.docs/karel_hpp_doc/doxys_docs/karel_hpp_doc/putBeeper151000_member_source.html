void Robot::putBeeper()
{
    int  x;

    if ( world != nullptr )
    {
       if ( beepers <= 0 )
       {
          show_Error ( "ERROR: No beeper." );
       }
       else
       {
          x = world->has_slot ( avenue, street );
          if ( x < 0 )
          {
             show_Error ( "ERROR: No slot free." );
          }
          else
          {
             beepers = beepers - 1;
             world->set ( avenue, street, '1' );
             debug ( );
          }
       }
    }
} // end Robot::putBeeper ( )
