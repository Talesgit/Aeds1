void Robot::move()
{
    if ( checkStatus ( ) &&
         frontIsClear ( ) )
    {
        switch ( direction )
        {
            case NORTH  :
              set (  0, +1 );
              break;
            case EAST  :
              set ( +1,  0 );
              break;
            case SOUTH :
              set (  0, -1 );
              break;
            case WEST  :
              set ( -1,  0 );
              break;
        }
    }
    else
    {
        show_Error ( "ERROR: Can't move." );
    }
} // end Robot::move ( )
